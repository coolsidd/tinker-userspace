139 int request_memory() {
140 
141  // increment current program break location
142  void *curr_brk = sbrk(0);
143  int ret = brk(curr_brk + INCREMENT);
144  if (ret == -1)
145  return -1; // failed to find program break location
146 
147  // allocate new arena of memory
148  Block_hdr hdr = (Block_hdr)curr_brk;
149  hdr->size = INCREMENT - sizeof(block_hdr);
150  hdr->allocated = false;
151  hdr->block_start = true;
152  LL_TAIL(&head, ((struct llhead *)hdr));
153 
154  return 0; // successfully allocated memory
